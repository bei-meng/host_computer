注释使用:#
立即数只支持：16进制数格式0x10或者0X10, 10进制数格式10101,10进制数最高位不能为0

addi a, b, 3                            # a = b + 3
bge a, b, label                         # if a>=b 就跳转到对应标签指令处
exit                                    # 结束指令
load_din_ram a,b                        # b里面存的是din_ram_addr,将din_ram对应位置的数据加载到a
add a,b,c                               # a = b + c
sub a,b,c                               # a = b - c
xori a,b,1                              # a = b ^ 1
sll a,b,c                               # a = b << c
srl a,b,c                               # a = b >> c
set_row_bank a,b                        # a为row_bank_mask,b为row_index_mask
set_col_bank a,b                        # a为col_bank_mask,b为col_index_mask
read_row_pulse_tia a,b,c                # a为tia的编号,b为存储的dout_ram地址,c为哪一块dout_ram
read_col_pulse_tia a,b,c                # a为tia的编号,b为存储的dout_ram地址,c为哪一块dout_ram
return_dout a,b,c                       # a为要返回的数据长度,b为dout_ram读取的起始地址,c为哪一块dout_ram
